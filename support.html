currentUser.uid : null, status: 'new', submittedAt: new Date().toISOString(), ticketId: 'TICKET-' + Date.now() + '-' + Math.random().toString(36).substr(2, 6).toUpperCase() }; // Validate data if (!complaintData.fullName || !complaintData.email || !complaintData.phone || !complaintData.subject || !complaintData.message) { showNotification('Please fill in all required fields', 'error'); return; } // Validate email const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/; if (!emailRegex.test(complaintData.email)) { showNotification('Please enter a valid email address', 'error'); return; } // Disable submit button submitBtn.disabled = true; submitBtn.innerHTML = ' Submitting...'; try { console.log("Submitting complaint:", complaintData); // Save to Firestore - complaints collection const complaintRef = await db.collection('complaints').add(complaintData); console.log("Complaint saved to Firestore with ID:", complaintRef.id); // Save to Realtime Database for admin access const rtdbRef = rtdb.ref(`complaints/${complaintData.ticketId}`); await rtdbRef.set({ ...complaintData, firestoreId: complaintRef.id }); console.log("Complaint saved to Realtime Database"); // Show success modal successModal.classList.add('active'); // Reset form (but keep user info) document.getElementById('subject').value = ''; document.getElementById('message').value = ''; } catch (error) { console.error('Error submitting complaint:', error); showNotification('Failed to submit complaint. Please try again.', 'error'); // Try localStorage as fallback console.log("Trying localStorage fallback..."); const localComplaints = JSON.parse(localStorage.getItem('complaints') || '[]'); localComplaints.push({ ...complaintData, localSave: true, savedAt: new Date().toISOString() }); localStorage.setItem('complaints', JSON.stringify(localComplaints)); showNotification('Complaint saved locally. We will sync when online.', 'info'); successModal.classList.add('active'); } finally { // Re-enable submit button submitBtn.disabled = false; submitBtn.innerHTML = ' Submit Complaint'; } }); // Close modal function closeModal() { successModal.classList.remove('active'); } // Notification function function showNotification(message, type) { // Remove existing notification const existing = document.querySelector('.custom-notification'); if (existing) existing.remove(); // Create notification const notification = document.createElement('div'); notification.className = 'custom-notification'; notification.style.cssText = ` position: fixed; top: 20px; right: 20px; z-index: 9999; animation: fadeInUp 0.3s ease; `; const bgColor = type === 'success' ? 'rgba(16, 185, 129, 0.9)' : type === 'error' ? 'rgba(239, 68, 68, 0.9)' : 'rgba(59, 130, 246, 0.9)'; notification.innerHTML = `
${message}
`; document.body.appendChild(notification); // Remove after 3 seconds setTimeout(() => { if (notification.parentNode) { notification.style.animation = 'fadeOut 0.3s ease'; setTimeout(() => notification.remove(), 300); } }, 3000); } // Add fadeOut animation const style = document.createElement('style'); style.textContent = ` @keyframes fadeOut { from { opacity: 1; transform: translateY(0); } to { opacity: 0; transform: translateY(-10px); } } `; document.head.appendChild(style); // Close modal when clicking outside successModal.addEventListener('click', (e) => { if (e.target === successModal) { closeModal(); } }); // Initialize document.addEventListener('DOMContentLoaded', initPage);
